exports.event_processor = (s) => {
    const streams = processors.map((processor) => {
        const byProcessor = event => event.ns == processor.name; 
        return s.fork()
                .filter(byProcessor)
                .through(processor.processor);
    });

    return highland(streams).merge();
};

exports.processors = processors;
